```
│   │   ├── app

│   │   └── index.js
│   ├── images
│   │   └── logo.svg
│   ├── packages
│   │   └── ...
│   ├── utils
│   │   ├── ...
│   │   └── index.js

```

Following structure proposed by: [Charles Stover article](https://medium.com/@Charles_Stover/optimal-file-structure-for-react-applications-f3e35ad0a145)
